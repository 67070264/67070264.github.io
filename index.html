      //
//  Arduino Sketch for reading Temperature and Humidity from a DHT22 Sensor
//  This code is designed for microcontrollers like Arduino Uno, ESP32, etc.
//  It reads the data and prints it to the Serial Monitor.
//

// 1. Include necessary libraries
// You need to install the "DHT sensor library" by Adafruit from the Library Manager.
#include 
#include 
#include 

// 2. Define constants for the sensor pin and type
#define DHTPIN 2         // The digital pin connected to the DHT sensor
#define DHTTYPE DHT22    // Define the type of sensor we are using (DHT22 or AM2302)

// 3. Initialize the DHT sensor object
DHT dht(DHTPIN, DHTTYPE);

// =========================================================================
// SETUP FUNCTION - Runs only once when the board starts up
// =========================================================================
void setup() {
  // Start serial communication at 9600 bits per second for debugging
  Serial.begin(9600);
  Serial.println(F("DHT22 Sensor Test - Initializing..."));

  // Initialize the sensor
  dht.begin();
}

// =========================================================================
// LOOP FUNCTION - Runs over and over again forever
// =========================================================================
void loop() {
  // Wait a few seconds between measurements.
  // The DHT22 sensor should not be read more than once every 2 seconds.
  delay(2000);

  // Read Humidity from the sensor
  float humidity = dht.readHumidity();
  // Read Temperature in Celsius (the default)
  float temp_c = dht.readTemperature();
  // Read Temperature in Fahrenheit
  float temp_f = dht.readTemperature(true);

  // Check if any reads failed and exit early to try again.
  if (isnan(humidity) || isnan(temp_c) || isnan(temp_f)) {
    Serial.println(F("Failed to read from DHT sensor!"));
    return; // Exit the function and try again in the next loop
  }

  // If readings were successful, print them to the Serial Monitor
  Serial.print(F("Humidity: "));
  Serial.print(humidity);
  Serial.print(F("%  |  "));
  Serial.print(F("Temperature: "));
  Serial.print(temp_c);
  Serial.print(F(" °C  ~  "));
  Serial.print(temp_f);
  Serial.println(F(" °F"));
}
